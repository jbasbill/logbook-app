// ── BLE Connection ──

async function connectDevice() {

  if (!navigator.bluetooth) {

    toast('Web Bluetooth not supported. Use Chrome on Android.');

    return;

  }

  const btn = document.getElementById('connect-btn');

  const status = document.getElementById('connect-status');

  const dot = document.getElementById('status-dot');


  btn.disabled = true;

  dot.className = 'status-dot connecting';

  status.textContent = 'Scanning for Logbook device...';


  try {

    device = await navigator.bluetooth.requestDevice({

      filters: [{ name: 'LOGBOOK' }],

      optionalServices: [SERVICE_CONFIG, SERVICE_DATA]

    });


    status.textContent = 'Connecting...';

    device.addEventListener('gattserverdisconnected', onDisconnected);


    server = await device.gatt.connect();

    status.textContent = 'Discovering services...';


    configService = await server.getPrimaryService(SERVICE_CONFIG);

    dataService   = await server.getPrimaryService(SERVICE_DATA);


    // Read firmware version

    try {

      const fwChar = await configService.getCharacteristic(CHAR_FIRMWARE);

      const fwVal  = await fwChar.readValue();

      const fw     = new TextDecoder().decode(fwVal);

      document.getElementById('firmware-badge').textContent = 'Firmware ' + fw;

      document.getElementById('firmware-badge').classList.add('show');

      document.getElementById('fw-version-display').textContent = fw;

    } catch(e) { /* firmware char not yet implemented — ok */ }


    // Read current field mask

    try {

      const fieldChar = await configService.getCharacteristic(CHAR_FIELDS);

      const fieldVal  = await fieldChar.readValue();

      fieldMask = fieldVal.getUint16(0, true);

    } catch(e) { fieldMask = 0x3FFF; }


    // Subscribe to session stats

    try {

      const sessChar = await dataService.getCharacteristic(CHAR_SESSION);

      await sessChar.startNotifications();

      sessChar.addEventListener('characteristicvaluechanged', onSessionUpdate);

    } catch(e) { /* not yet implemented */ }


    // Subscribe to device status

    try {

      const statChar = await dataService.getCharacteristic(CHAR_STATUS);

      await statChar.startNotifications();

      statChar.addEventListener('characteristicvaluechanged', onStatusUpdate);

    } catch(e) { /* not yet implemented */ }


    onConnected();


  } catch(e) {

    btn.disabled = false;

    dot.className = 'status-dot';

    if (e.name !== 'NotFoundError') {

      status.textContent = 'Connection failed: ' + e.message;

    } else {

      status.textContent = 'No device selected.';

    }

  }

}


function onConnected() {

  const dot = document.getElementById('status-dot');

  dot.className = 'status-dot connected';

  document.getElementById('connect-status').textContent = 'Connected to ' + device.name;

  document.getElementById('connect-btn').textContent = 'Connected';

  document.getElementById('connect-btn').disabled = true;

  document.getElementById('device-name-display').textContent = device.name;

  document.getElementById('dash-connected').style.display = 'flex';

  document.getElementById('dash-disconnected').style.display = 'none';

  toast('Connected to ' + device.name);

}


function onDisconnected() {

  device = null; server = null; configService = null; dataService = null;

  document.getElementById('status-dot').className = 'status-dot';

  document.getElementById('connect-btn').textContent = 'Connect';

  document.getElementById('connect-btn').disabled = false;

  document.getElementById('connect-status').textContent = 'Disconnected.';

  document.getElementById('firmware-badge').classList.remove('show');

  document.getElementById('dash-connected').style.display = 'none';

  document.getElementById('dash-disconnected').style.display = 'flex';

  toast('Disconnected');

}


async function disconnectDevice() {

  if (device && device.gatt.connected) device.gatt.disconnect();

}


// ── Live data handlers ──

function onSessionUpdate(e) {

  try {

    const json = JSON.parse(new TextDecoder().decode(e.target.value));

    document.getElementById('stat-distance').textContent = (json.d || 0).toFixed(1);

    document.getElementById('stat-rpm').textContent = (json.rpm || 0).toLocaleString();

    document.getElementById('stat-speed').textContent = Math.round(json.spd || 0);

    const dur = json.dur || 0;

    const h = Math.floor(dur / 3600);

    const m = Math.floor((dur % 3600) / 60);

    const s = dur % 60;

    document.getElementById('stat-duration').textContent =

      h > 0 ? h+':'+String(m).padStart(2,'0')+':'+String(s).padStart(2,'0')

             : m+':'+String(s).padStart(2,'0');

  } catch(e) {}

}

  const byte = e.target.value.getUint8(0);

  document.getElementById('pill-recording').classList.toggle('active', !!(byte & 1));

  document.getElementById('pill-printing').classList.toggle('active',  !!(byte & 2));

  document.getElementById('pill-obd').classList.toggle('active',       !!(byte & 4));

  document.getElementById('pill-gps').classList.toggle('active',       !!(byte & 8));

}

    const ch = await configService.getCharacteristic(CHAR_REPRINT);

// ── Slip editor ──

function renderEditor() {

  const container = document.getElementById('field-toggles');

  container.innerHTML = '';

  FIELDS.forEach(f => {

    const on = !!(fieldMask & (1 << f.bit));

    const row = document.createElement('div');

    row.className = 'field-row';

    row.innerHTML = `

      <div>

        <div class="field-name">${f.label}</div>

        <div class="field-sub">${f.sub}</div>

      </div>

      <div class="toggle ${on ? 'on' : ''}" data-bit="${f.bit}" onclick="toggleField(this)"></div>

    `;

    container.appendChild(row);

  });

  renderSlipPreview();

}


function toggleField(el) {

  const bit = parseInt(el.dataset.bit);

  el.classList.toggle('on');

  if (el.classList.contains('on')) {

    fieldMask |= (1 << bit);

  } else {

    fieldMask &= ~(1 << bit);

  }

  markDirty();

  renderSlipPreview();

}

  if (fieldMask & (1 << 3))  lines.push('Avg speed:  42.7 mph');

  if (fieldMask & (1 << 4))  lines.push('Top speed:  81.0 mph');

  lines.push('------------------------');

  if (fieldMask & (1 << 5))  lines.push('Peak RPM:      4,820');

  if (fieldMask & (1 << 6))  lines.push('Avg throttle:    28%');

  if (fieldMask & (1 << 7))  lines.push('Max load:        74%');

  if (fieldMask & (1 << 8))  lines.push('Coolant:        192F');

  if (fieldMask & (1 << 9))  lines.push('Boost:        14.2psi');

  if (fieldMask & (1 << 10)) lines.push('Battery:        12.6V');

  lines.push('------------------------');

  if (fieldMask & (1 << 11)) lines.push('Start: 37.771N 122.431W');

  if (fieldMask & (1 << 12)) lines.push('End:   37.802N 122.418W');

  if (fieldMask & (1 << 13)) lines.push('RPM [========----] 82%');

}


function markDirty() { dirty = true; }


async function saveConfig() {

  if (!configService) {

    // Save locally even without connection

    localStorage.setItem('fieldMask', fieldMask);

    localStorage.setItem('units', document.getElementById('units-select').value);

    localStorage.setItem('vehicleName', document.getElementById('vehicle-name').value);

    toast('Saved locally (device not connected)');

    dirty = false;

    return;

  }

  try {

    // Write field mask

    const fieldCh = await configService.getCharacteristic(CHAR_FIELDS);

    const buf = new ArrayBuffer(2);

    new DataView(buf).setUint16(0, fieldMask, true);

    await fieldCh.writeValueWithResponse(buf);


    // Write units

    const unitsCh = await configService.getCharacteristic(CHAR_UNITS);

    await unitsCh.writeValueWithResponse(

      new Uint8Array([parseInt(document.getElementById('units-select').value)])

    );


    // Write vehicle name

    const nameCh = await configService.getCharacteristic(CHAR_VEHICLE);

    const nameBytes = new TextEncoder().encode(document.getElementById('vehicle-name').value);

    await nameCh.writeValueWithResponse(nameBytes);


    // Save locally too

    localStorage.setItem('fieldMask', fieldMask);

    localStorage.setItem('units', document.getElementById('units-select').value);

    localStorage.setItem('vehicleName', document.getElementById('vehicle-name').value);


    toast('Saved to device ✓');

    dirty = false;

  } catch(e) {

    toast('Save failed: ' + e.message);

  }

}


// ── History ──

function renderHistory() {

  const trips = getTripsFromDB();

  const list = document.getElementById('history-list');

  const empty = document.getElementById('history-empty');

  if (trips.length === 0) {

    empty.style.display = 'flex';

    return;

  }

  empty.style.display = 'none';

  list.innerHTML = '';

  trips.sort((a,b) => b.ts - a.ts).forEach(trip => {

    const el = document.createElement('div');

    el.className = 'trip-item';

    const date = new Date(trip.ts * 1000);

    const dateStr = date.toLocaleDateString('en-US', {weekday:'short', month:'short', day:'numeric', year:'numeric'});

    el.innerHTML = `

      <div class="trip-head">

        <span class="trip-date">${dateStr}</span>

        <span class="trip-dist">${(trip.d||0).toFixed(1)} mi</span>

      </div>

      <div class="trip-stats">

        <span class="trip-stat">${formatDuration(trip.dur)}</span>

        <span class="trip-stat">${(trip.rpm||0).toLocaleString()} RPM peak</span>

        <span class="trip-stat">${Math.round(trip.spd||0)} mph top</span>

      </div>

    `;

    list.appendChild(el);

  });

}


function formatDuration(s) {

  const h = Math.floor(s/3600), m = Math.floor((s%3600)/60);

  return h > 0 ? h+'h '+m+'m' : m+'m';

}


// ── Simple localStorage trip store (IndexedDB upgrade later) ──

function getTripsFromDB() {

  try { return JSON.parse(localStorage.getItem('trips') || '[]'); } catch(e) { return []; }

}


function saveTripToDB(trip) {

  const trips = getTripsFromDB();

  trips.push(trip);

  localStorage.setItem('trips', JSON.stringify(trips));

}

}

  }

}
